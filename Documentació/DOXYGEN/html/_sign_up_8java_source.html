221  String u = user.getText();
222  String psw = new String (pass.getPassword());
223  String cfrm = new String (confirm.getPassword());
224 
225  if (u.isBlank()) {
226  user.setBackground(new Color(234, 95, 95));
227  pass.setBackground(new Color(234, 95, 95));
228  confirm.setBackground(new Color(234, 95, 95));
229  user.setText("");

231  confirm.setText("");
232  missatge.setText("*User incorrect*");
233 
234  }
235  else if (cp.existeixUsuari(u)) {
236  user.setBackground(new Color(234, 95, 95));
237  pass.setBackground(new Color(234, 95, 95));
238  confirm.setBackground(new Color(234, 95, 95));

240  pass.setText("");
241  confirm.setText("");
242  missatge.setText("*User already exists*");
243  }
244 
245  else if (psw.equals(cfrm)) {
246  missatge.setText("");
247  cp.signUp(u, psw);
248  MenuPrincipal menu = new MenuPrincipal(parent,cp);
249  menu.setVisible(true);
250  dispose();
251  }
252

254  user.setBackground(new Color(234, 95, 95));
